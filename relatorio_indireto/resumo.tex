\section{Resumo das equações do método}

Abaixo, resumimos algumas das principais equações utilizadas no método.

\vspace{20mm}

\equacao{Planta}
  {\dot{y}_p = a_p y_p + u \label{eq:planta}}

\equacao{Modelo}
  {\dot{y}_m = -a_m y_m + r \label{eq:ref_model}}

\equacao{Erro de saída}
  {e_0 = y_p - y_m \label{eq:error}}

\equacao{Lei de controle}
  {u(t) = k(t) \, y_p + r \label{eq:ctrl_law}}

\equacao{Filtro}
  { F(s) = \frac{1}{s+a_f} \label{eq:filter}}

%\equacao{Ganho ideal (matching gain)}
  %{k^* = -a_p-a_m \label{eq:ideal_gain}}

\equacao{Ganho estimado}
  {k(t) = -\hat{a}_p-a_m \label{eq:est_gain}}

\equacao{Saída filtrada}
	{\phi = \mathcal{L}^{-1}\left\{ \frac{y}{s+a_f} \right\}}

\equacao{Parâmetro estimado}
	{\theta = a_f + \hat{a}_p}

\equacao{Predição da saída}
	{\hat{y} = \theta \, \phi + \mathcal{L}^{-1}\left\{ \frac{u}{s+a_f} \right\} \label{eq:pred_saida} }

\equacao{Erro de Predicao}
	{\epsilon = \hat{y} - y \label{eq:pred_error}}

\equacao{Lei de adaptação}
  {\dot{\theta} = - \frac{\gamma \epsilon \phi}{m^2} \,, \quad m^2 = 1+\phi^2 \label{eq:adpt_law}}
